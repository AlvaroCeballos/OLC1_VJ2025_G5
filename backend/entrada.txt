INT m = 0;

DO {
    INT n = 0;
    DO {
        IF (n == 1 && m == 1) {
            n = n + 1;
            CONTINUE;
        }
        PRINTLN("m = " + m + ", n = " + n);
        IF (m == 2 && n == 0) {
            BREAK;
        }
        n = n + 1;
    } WHILE (n < 3);
    m = m + 1;
} WHILE (m < 3);
